\documentclass[margin=5pt]{standalone}
\usepackage{tikz}

\usepackage[T1]{fontenc}

% The 'sfdefault' option to make the base font sans serif.
\usepackage[sfdefault]{AlegreyaSans}
\renewcommand*\oldstylenums[1]{{\AlegreyaSansOsF #1}}

% Radius of circle circumscribing the hexagons.
\pgfmathsetmacro{\rad}{5.75}
% Horizontal distance between the centers of the hexagons.
\pgfmathsetmacro{\hdist}{1.7 * \rad}
% Line width for the hexagons.
\pgfmathsetmacro{\lw}{1}

% Draw a hexagon.
\newcommand{\hexagon}[1]{
    \draw[orange, line width=\lw mm, rounded corners=\lw mm]
        (0:\rad mm)--(60:\rad mm)--(120:\rad mm)--(180:\rad mm)--
        (240:\rad mm)--(300:\rad mm)--cycle;
        \node{#1};
}


\begin{document}

\begin{tikzpicture}
    {\Huge
    \hexagon{A}
    \begin{scope}[xshift=\hdist mm, yshift=-\rad mm]\hexagon{S}\end{scope}
    \begin{scope}[xshift=2*\hdist mm]\hexagon{I}\end{scope}
    \begin{scope}[xshift=3*\hdist mm, yshift=-\rad mm]\hexagon{X};\end{scope}
    }
    \node at (1.5, -1.35) {\small Analog Spectral Imager for X-rays};
\end{tikzpicture}

\end{document}
